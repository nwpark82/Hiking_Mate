// hooks/useGPSTracking.ts
function useGPSTracking(hikeId: string) {
  const [isTracking, setIsTracking] = useState(false)
  const [position, setPosition] = useState<Position | null>(null)
  const [path, setPath] = useState<Position[]>([])
  
  useEffect(() => {
    if (!isTracking) return
    
    const watchId = navigator.geolocation.watchPosition(
      (pos) => {
        const newPoint = {
          latitude: pos.coords.latitude,
          longitude: pos.coords.longitude,
          altitude: pos.coords.altitude,
          accuracy: pos.coords.accuracy,
          timestamp: Date.now()
        }
        
        // ì •í™•ë„ í•„í„°ë§
        if (newPoint.accuracy <= 30) {
          setPosition(newPoint)
          setPath(prev => [...prev, newPoint])
          
          // IndexedDBì— ì €ì¥
          saveToIndexedDB(hikeId, newPoint)
        }
      },
      (error) => console.error('GPS error:', error),
      {
        enableHighAccuracy: true,
        timeout: 10000,
        maximumAge: 0
      }
    )
    
    return () => navigator.geolocation.clearWatch(watchId)
  }, [isTracking, hikeId])
  
  // í†µê³„ ê³„ì‚°
  const stats = useMemo(() => {
    if (path.length < 2) return null
    
    return {
      distance: calculateTotalDistance(path),
      duration: path[path.length - 1].timestamp - path[0].timestamp,
      avgSpeed: calculateAvgSpeed(path),
      maxAltitude: Math.max(...path.map(p => p.altitude || 0))
    }
  }, [path])
  
  return {
    isTracking,
    startTracking: () => setIsTracking(true),
    pauseTracking: () => setIsTracking(false),
    stopTracking: () => {
      setIsTracking(false)
      return { path, stats }
    },
    position,
    path,
    stats
  }
}
